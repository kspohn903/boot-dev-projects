def edit_distance(str1, str2):
    """
    Calculates the Levenshtein distance between two strings using dynamic programming.

    Args:
        str1 (str): The first string.
        str2 (str): The second string.

    Returns:
        int: The minimum number of edits (insertions, deletions, or substitutions)
             required to change str1 into str2.
    """
    # Get the lengths of the two strings.
    n1, n2 = len(str1), len(str2)

    # Create a 2D table (list of lists) to store the distances.
    # The table size will be (n1 + 1) x (n2 + 1).
    dp = [[0] * (n1 + 1) for _ in range(n2 + 1)]

    # Initialize the first row and column of the table.
    # The distance from an empty string to a string of length i is i.
    for i in range(n2 + 1):
        dp[i][0] = i
        for j in range(n1 + 1):
            dp[0][j] = j

    # Fill the rest of the table using the dynamic
    # programming approach.
    for i in range(1, n1 + 1):
        for j in range(1, n2 + 1):
            # Calculate the cost of a substitution. 
            # If the characters are the same,
            # the cost is 0; otherwise, it's 1.
            cost = 0 if str1[i-1] == str2[j-1] else 1
            
            # The current cell's value is the 
            # minimum of three possibilities:
            # 1. Deletion:  dp[i-1][j] + 1
            # 2. Insertion: dp[i][j-1] + 1
            # 3. Substitution: dp[i-1][j-1] + cost
            dp[i][j] = min(dp[i-1][j] + 1, # Deletion
                           dp[i][j-1] + 1, # Insertion
                           dp[i-1][j-1] + cost) 
                           # Substitution
    # The value in the bottom-right corner of the table 
    # is the Levenshtein distance.
    return dp[n1][n2]
