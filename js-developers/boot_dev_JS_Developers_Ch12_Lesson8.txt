async function return_message() {
  let message = await updateMessageStatus();
  console.log(`${message}`);
  return message;
  // don't touch below this line
}

function updateMessageStatus(messageId, currentStatus, isDelivered) { 
  return new Promise((resolve, reject) => {
    setTimeout(() => {
      let message = "";
      if (currentStatus === "Sending" && isDelivered) {
          message = `Textio Message ${messageId} has been delivered successfully.`;
          resolve(message);
        } else if (currentStatus === "Sending" && !isDelivered) {
          message = `Textio Message ${messageId} is still sending and cannot be marked as delivered.`
          reject(message);
      } else {
        message = `Textio Message ${messageId} status updated to ${currentStatus}.`;
        resolve(message);
      }
    }, 1000);
  });
}
